\T{Por que $(-) \times (-) = (+)$?}

Observe o padrĂŁo da sequĂȘncia de $n \times (-2)$ Ă  medida que $n$ decresce:

\vspace{4pt}
$3 \times (-2) = -6$

$2 \times (-2) = -4$

$1 \times (-2) = -2$

$0 \times (-2) = \phantom{+}0$

$-1 \times (-2) = +2$ \quad $\leftarrow$ o padrĂŁo exige que suba $+2$

$-2 \times (-2) = +4$ \quad $\leftarrow$ negativo $\times$ negativo $=$ positivo \checkmark

\vspace{4pt}
\textbf{InterpretaĂ§ĂŁo:} multiplicar por $-1$ inverte o sinal. Inverter duas vezes retorna ao original: $(-1)\times(-1)\times a = +a$. O negativo do negativo Ă© o positivo.

\T{Sinais Iguais $\Rightarrow$ Resultado Positivo}

A multiplicaĂ§ĂŁo ou divisĂŁo de dois nĂșmeros com \textbf{sinais iguais} resulta sempre em valor \textbf{positivo}.

\Formula{(+)\times(+)=(+) \qquad (-)\times(-)=(+)}
\Formula{(+)\div(+)=(+) \qquad (-)\div(-)=(+)}

\TM{Exemplos}

$(+7)\times(+3) = +21$

$(+15)\div(+3) = +5$

$(-7)\times(-3) = +21$

$(-15)\div(-3) = +5$

\vspace{4pt}
\textbf{Regra prĂĄtica:} sinais iguais $\Rightarrow$ resultado \emph{positivo}.

\T{Sinais Diferentes $\Rightarrow$ Resultado Negativo}

A multiplicaĂ§ĂŁo ou divisĂŁo de dois nĂșmeros com \textbf{sinais diferentes} resulta sempre em valor \textbf{negativo}.

\Formula{(+)\times(-)=(-) \qquad (-)\times(+)=(-)}
\Formula{(+)\div(-)=(-) \qquad (-)\div(+)=(-)}

\TM{Exemplos}

$(+7)\times(-3) = -21$

$(+15)\div(-3) = -5$

$(-7)\times(+3) = -21$

$(-15)\div(+3) = -5$

\vspace{4pt}
\textbf{Regra prĂĄtica:} sinais diferentes $\Rightarrow$ resultado \emph{negativo}.

\T{Algoritmo de CĂĄlculo}

\textbf{Passo 1 â} Determine o sinal do resultado pela regra dos sinais.

\textbf{Passo 2 â} Multiplique (ou divida) os valores absolutos normalmente.

\textbf{Passo 3 â} Aplique o sinal ao resultado.

\TM{Exemplo 1 â $(-12)\times(+5)$}

Sinais diferentes $\Rightarrow$ negativo. $\;12\times5=60$. \quad Resultado: $-60$ \checkmark

\TM{Exemplo 2 â $(-48)\div(-6)$}

Sinais iguais $\Rightarrow$ positivo. $\;48\div6=8$. \quad Resultado: $+8$ \checkmark

\TM{Exemplo 3 â $(-3)\times(+7)$}

Sinais diferentes $\Rightarrow$ negativo. $\;3\times7=21$. \quad Resultado: $-21$ \checkmark

\T{MĂșltiplos Fatores}

Em um produto com trĂȘs ou mais fatores, conte a quantidade de fatores negativos:

\textbf{Quantidade par de negativos $\Rightarrow$ resultado positivo.}

\textbf{Quantidade Ă­mpar de negativos $\Rightarrow$ resultado negativo.}

\TM{Exemplos}

$(-2)\times(-3)\times(-4)$: \textbf{3 negativos} (Ă­mpar) $\Rightarrow$ $-(2\times3\times4) = -24$ \checkmark

$(-1)\times(-2)\times(-3)\times(-4)$: \textbf{4 negativos} (par) $\Rightarrow$ $+(1\times2\times3\times4) = +24$ \checkmark

$(+5)\times(-2)\times(+3)\times(-1)$: \textbf{2 negativos} (par) $\Rightarrow$ $+30$ \checkmark

\T{PotĂȘncias de Inteiros Negativos}

\Formula{(-a)^n = \begin{cases} +a^n & \text{se } n \text{ Ă© par} \\ -a^n & \text{se } n \text{ Ă© Ă­mpar} \end{cases}}

\TM{Exemplos}

$(-3)^2 = (-3)\times(-3) = +9$ \quad (par $\Rightarrow$ positivo)

$(-3)^3 = (-3)\times(-3)\times(-3) = -27$ \quad (Ă­mpar $\Rightarrow$ negativo)

$(-2)^4 = +16$ \quad $(-2)^5 = -32$ \quad $(-1)^{101} = -1$

\vspace{4pt}
\textbf{AtenĂ§ĂŁo â o parĂȘntese faz diferenĂ§a:}

$(-2)^2 = +4$ \quad (eleva $-2$ ao quadrado)

$-2^2 = -4$ \quad (eleva $2$ ao quadrado e aplica $-$ depois)

\T{Propriedades da MultiplicaĂ§ĂŁo}

\Formula{a \times b = b \times a \quad \text{(comutativa)}}
\Formula{(a\times b)\times c = a\times(b\times c) \quad \text{(associativa)}}
\Formula{a\times(b+c) = a\times b + a\times c \quad \text{(distributiva)}}
\Formula{a \times 1 = a \quad \text{(elemento neutro)}}
\Formula{a \times 0 = 0 \quad \text{(elemento absorvente)}}
\Formula{(-1)\times a = -a \quad \text{(inversor de sinal)}}

\TM{Exemplo â distributiva}

$(-3)\times(4+2) = (-3)\times4 + (-3)\times2 = -12+(-6) = -18$

VerificaĂ§ĂŁo: $(-3)\times6 = -18$ \checkmark

\T{RelaĂ§ĂŁo $\times$ e $\div$ â Prova Real}

\Formula{\text{Se } a\times b = c \text{, entĂŁo } c\div b = a \text{ e } c\div a = b}

\TM{Exemplos}

$(-4)\times(+5) = -20$

Prova: $(-20)\div(+5) = -4$ \checkmark \quad e \quad $(-20)\div(-4) = +5$ \checkmark

\vspace{4pt}
$(-6)\times(-7) = +42$

Prova: $(+42)\div(-7) = -6$ \checkmark \quad e \quad $(+42)\div(-6) = -7$ \checkmark

\T{Erros Comuns}
